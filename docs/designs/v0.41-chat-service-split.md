**职责：** agent 循环 + 流式输出 + 会话管理

```python
class ChatOrchestrator:
    """Agent chat orchestrator — the main entry point for chat()."""
    
    def __init__(
        self,
        wiki_service: Any,
        chat_db: ChatDatabase,
        memory_manager: Any = None,
        config: dict | None = None,
    ):
        self.wiki_service = wiki_service
        self.db = chat_db
        self.memory_manager = memory_manager
        self.config = config or merge_six_step_config()
        
        # Delegate to specialized components
        self.prompt_builder = PromptBuilder(wiki_service, memory_manager)
        self.context_manager = ContextManager(config=self.config)
        self.tool_executor = ToolExecutor(
            chat_db=chat_db,
            memory_manager=memory_manager,
            config=self.config,
        )
        
        # State
        self._session_status: dict[str, str] = {}
        self._abort_events: dict[str, asyncio.Event] = {}
    
    async def chat(
        self, message: str, session_id: str | None = None,
        wiki_id: str | None = None, jwt_token: str | None = None,
    ) -> AsyncIterator[dict]:
        """Stream a chat response as SSE events."""
        # ... agent loop
    
    async def delete_session(self, session_id: str) -> dict:
        self.context_manager.remove(session_id)
        self.db.delete_chat_session(session_id)
        return {"deleted": True}
    
    async def abort_session(self, session_id: str) -> dict:
        if session_id in self._abort_events:
            self._abort_events[session_id].set()
        return {"aborted": True, "session_id": session_id}
```

### 3.2 PromptBuilder

**位置：** `src/llmwikify/apps/chat/agent/prompt_builder.py`

**职责：** 系统提示组合

```python
class PromptBuilder:
    """Builds the system prompt from multiple sections."""
    
    def __init__(self, wiki_service: Any, memory_manager: Any = None):
        self.wiki_service = wiki_service
        self.memory_manager = memory_manager
    
    async def build(
        self, wiki_id: str | None = None,
        user_message: str | None = None,
        session_id: str | None = None,
    ) -> str:
        """Build the complete system prompt."""
        parts = []
        parts.append(self._role_section())
        if wiki_id:
            parts.append(self._wiki_context_section(wiki_id))
        prefs = await self._preferences_section()
        if prefs:
            parts.append(prefs)
        tools = await self._tools_section()
        if tools:
            parts.append(tools)
        parts.append(self._date_section())
        related = await self._related_section(user_message, session_id)
        if related:
            parts.append(related)
        return "\n\n".join(parts)
    
    @staticmethod
    def parse_wiki_prefix(message: str) -> tuple[str | None, str]:
        """Parse @wiki_id prefix from message."""
        match = re.match(r"^@(\S+)\s+(.*)$", message, re.DOTALL)
        if match:
            return match.group(1), match.group(2)
        return None, message
```

### 3.3 ContextManager

**位置：** `src/llmwikify/apps/chat/agent/context_manager.py`

**职责：** 运行时状态 + token 截断 + 压缩

```python
class ContextManager:
    """Manages in-memory conversation state for active sessions."""
    
    def __init__(self, config: dict | None = None):
        self.config = config or merge_six_step_config()
        self._contexts = ContextStore(
            max_size=self.config.get("context_store_max_size", 200),
            ttl_seconds=self.config.get("context_store_ttl_seconds", 1800),
        )
    
    async def get_or_create(
        self, session_id: str, wiki_id: str | None,
        history_loader: Callable,
    ) -> AgentContext:
        """Get existing context or create new one."""
    
    async def prepare_messages(
        self, ctx: AgentContext, system_prompt: str, llm_client: Any,
    ) -> list[dict]:
        """Truncate and compact messages for LLM."""
        messages = [system_prompt_msg] + ctx.get_messages()
        messages = await self.compact(messages, llm_client)
        messages = self.truncate(messages, llm_client)
        return messages
    
    async def compact(
        self, messages: list[dict], llm_client: Any,
    ) -> list[dict]:
        """Summarize old messages when near capacity."""
    
    def truncate(
        self, messages: list[dict], llm_client: Any,
    ) -> list[dict]:
        """Hard-truncate to fit context window."""
    
    def _compute_budget(self, llm_client: Any) -> int:
        """Compute available token budget (shared by compact/truncate)."""
    
    def remove(self, session_id: str) -> None:
        self._contexts.remove(session_id)
```
